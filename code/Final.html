def rmsle(y_true, y_pred, y_type=None):
    n = len(y_true)
    y_true = np.array(y_true)
    y_pred = np.array(y_pred)
    parcels = [(np.log(y_pred[i] + 1) - np.log(y_true[i] + 1))**2 for i in xrange(n)]
    return np.sqrt(np.sum(parcels) / n)
